## GCD example
function @gcd:num (a:num, b:num)
{
    ## Everything divides 0
    when (a == 0) do return b;
    when (b == 0) do return a;
    when (a == b) do return a; ## base case

    if (a > b) do
        return @gcd(a-b, b);
    else
        return @gcd(a, b-a);
}
print @gcd(32, 24);
print @gcd(21, 13);
end
