## función min(x, y)
function @min:num (x:num, y:num)
{
    when ((x < y) == true) do return x;
    return y;
}
## función max(x, y)
function @max:num (x:num, y:num)
{
    if ((x < y) == false) do
    {
        return x;
    }
    else
    {
        return y;
    }
}
print @min(1,2);
print @max(1,2);
end
