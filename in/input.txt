var max:num, i:num;
max := 3;

unless (i == max) do {
    print ++i;
}

do {
    print ++i;
} until (i == 2 * max)

while (i != 3 * max) do {
    print ++i;
}

do {
    print ++i;
} while (i != 4 * max)

loop {
    print ++i;
    when (i == 5 * max) do break;
}

repeat 3 : {
    print ++i;
}
end
